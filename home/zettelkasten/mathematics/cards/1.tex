\zheader{Definición de espacio topológico}{1a1b}{propiedad universal, caracterización, suma topológica}


\begin{proposition}
Un conjunto $A$ es abierto si y sólo si $A^\circ = A$.
\end{proposition}
\begin{proof}
Trivial.
\end{proof}

\begin{proposition}
Un conjunto es cerrado si y sólo si $A = \overline{A}$.
\end{proposition}
