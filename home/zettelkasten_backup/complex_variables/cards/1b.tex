\zheader{Propiedades de campo}{1b}{campo, propiedades}

\begin{theorem}
Si $z, w \in \mathbb{C}$ y $n \in \mathbb{N}$, se verifican las siguientes propiedades
\begin{enumerate}[label=\textnormal{(\roman*)}]
\item (Binomio de Newton)
\begin{equation*}
    (z + w)^n = \sum_{k=0}^{n} \binom{n}{k} z^k w^{n-k}, \text{ donde } \binom{n}{k} = \frac{n!}{k! (n-k)!},
\end{equation*}
\item (Factorizaci√≥n de diferencia de potencias)
\begin{align*}
    z^n - w^n &= (z - w) \sum_{k=0}^{n-1} z^k w^{n-1-k} \\
              &= (z - w)(z^{n-1} + z^{n-2} w + \cdots + z w^{n-2} + w^{n-1}).
\end{align*}
\item (Suma de cuadrados) $z^2 + w^2 = (z + i w)(z - i w)$.
\end{enumerate}
\end{theorem}

\begin{proof}
({\scshape\romannumeral 1}) Se verifica en cualquier anillo siempre que $z w = w z$. ({\scshape\romannumeral 2}) Se verifica en cualquier anillo. ({\scshape\romannumeral 3}) $(z + i w)(z - i w) = z^2 - z i w + i w z - i^2 w = z^2 + w^2$, pues $z i w = i w z$ y $i^2 = - 1$.
\end{proof}
