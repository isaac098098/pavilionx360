\zheader{Relación entre argumentos de conjugados}{1d2}{argumento, conjugados}

\begin{proposition}
Si $z = a + i b \in \mathbb{C}$ es un número real negativo, entonces $\arg(\overline{z}) = \arg(z) = \pi$, y en caso contrario, $\arg(\overline{z}) = - \arg(z)$.
\end{proposition}

\end{proof}
